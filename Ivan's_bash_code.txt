#!/bin/bash

echo "Введите файл: "
read b

echo "Введите домен или адрес"
read a

echo "Поиск в файле"
c="$(grep $a $b)"

if [[ ! $c =~ $a ]]
then
echo "Успех! Ваши параметры не найдены"
else
echo "Угроза!!! Найдены следующие совпадения. необходимо срочно принять меры!!!"
grep $a $b
fi
